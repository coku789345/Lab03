void setup()
{
   Serial.begin(9600);
    Serial.println(Hello,World);
   pinMode(8,OUTPUT);
}
void loop()
{
      char a=Serial.read();
      if(a=='1')
      {
        digitalWrite(8,1);
        Serial.write(LED ONn);
      }
      else if(a=='0')
      {
        digitalWrite(8,0);
        Serial.write(LED OFFn);
      }
}
